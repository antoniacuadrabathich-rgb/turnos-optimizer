⚙ Restricciones Globales
📌 Regla automática: Máximo 6 días seguidos. Si un trabajador lleva 6 días consecutivos, el optimizador le asigna descanso obligatorio.
☀ Regla automática: Si trabaja un domingo, tiene garantizado al menos 1 día libre en los 6 días siguientes (lunes a sábado).
◷ Turnos Configurados
Cada turno tiene cobertura L-V y S-D independiente. El mínimo es la cobertura obligatoria (si no se cumple, se genera alerta). El máximo es hasta cuántos trabajadores asigna el optimizador si tienen horas disponibles. Usa 99 para rellenar al máximo.
🔗 Grupos Obligatorios — todos deben estar en el mismo turno
Los grupos se guardan mientras usas la app. El optimizador fuerza que todos los miembros queden en el mismo turno. Puedes además restringir qué turnos no puede hacer el grupo.

① Pegar el script en tu Google Sheet
Para escribir en tu Sheet sin OAuth necesitas un Google Apps Script como puente.
Sigue estos pasos una sola vez:
1 Abre tu Google Sheet → menú Extensiones → Apps Script
2 Borra todo el código existente y pega el código de abajo
3 Clic en Implementar → Nueva implementación → Tipo: Aplicación web
Ejecutar como: Yo · Acceso: Cualquier persona → Implementar
4 Copia la URL de implementación y pégala abajo
var SHEET_ID = '19pvbhBlDFIw0Bv7soFW2LBJN-8ikCfbVImXNLVTz12Q'; function doPost(e) { var data = JSON.parse(e.postData.contents); var ss = SpreadsheetApp.openById(SHEET_ID); var action = data.action; if (action === 'writeSchedule') { var sheet = ss.getSheetByName(data.sheetName); if (!sheet) sheet = ss.insertSheet(data.sheetName); sheet.clearContents(); sheet.getRange(1,1,data.rows.length,data.rows[0].length).setValues(data.rows); var hdr = sheet.getRange(1,1,1,data.rows[0].length); hdr.setBackground('#1a3a5c'); hdr.setFontColor('#ffffff'); hdr.setFontWeight('bold'); sheet.setFrozenRows(1); sheet.setFrozenColumns(2); return ContentService.createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } if (action === 'writeWorkers') { var sheet2 = ss.getSheetByName('Trabajadores'); if (!sheet2) sheet2 = ss.insertSheet('Trabajadores'); sheet2.clearContents(); sheet2.getRange(1,1,data.rows.length,data.rows[0].length).setValues(data.rows); var hdr2 = sheet2.getRange(1,1,1,data.rows[0].length); hdr2.setBackground('#1a3a5c'); hdr2.setFontColor('#ffffff'); hdr2.setFontWeight('bold'); return ContentService.createTextOutput(JSON.stringify({ok:true})) .setMimeType(ContentService.MimeType.JSON); } return ContentService.createTextOutput(JSON.stringify({error:'Unknown action'})) .setMimeType(ContentService.MimeType.JSON); } function doGet(e) { var ss = SpreadsheetApp.openById(SHEET_ID); var sheetName = e.parameter.sheet || 'Trabajadores'; var sheet = ss.getSheetByName(sheetName); if (!sheet) { return ContentService.createTextOutput(JSON.stringify({error:'Sheet not found'})) .setMimeType(ContentService.MimeType.JSON); } var values = sheet.getDataRange().getValues(); return ContentService.createTextOutput(JSON.stringify({values:values})) .setMimeType(ContentService.MimeType.JSON); }
